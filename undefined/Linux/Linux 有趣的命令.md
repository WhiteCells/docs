```sh
# 将输入的文本以 ASCII 艺术的形式显示在一个笑脸牛的图像中
sudo apt install cowsay
cowsay Hello, World!

# 输出随机的名言警句、笑话或其他有趣的文本
sudo apt install fortune
fortune

# 会显示一个火车在终端上通过
sudo apt install sl
sl

# 将输入的文本以 ASCII 艺术的形式显示出来
sudo apt install figlet
figlet Hello, World!

# 在终端中显示一个模拟的数字雨效果
sudo apt install cmatrix
cmatrix

# 将输入的文本以彩虹色的形式显示出来
sudo apt install lolcat
lolcat
```
