```sh
# .tar
tar -xvf <filename>.tar
tar -cvf <filename>.tar <dir> # 打包

# .tar.gz 或 .tgz
tar -xzvf <filename>.tar.gz
tar -czvf <filename>.tar.gz <dir>

# .tar.bz2
tar -xjvf <filename>.tar.bz2
tar -cjvf <filename>.tar.bz2 <dir>

# .tar.xz
tar -xJvf <filename>.tar.zx
tar -cJvf <filename>.tar.zx <dir>

# .gz
gunzip <filename>.gz
gzip <dir>

# .bz2
bunzip2 <filename>.bz2
bzip2 <dir>

# .xz
unxz <filename>.xz
xz <dir>

# .zip
unzip <filename>.zip
zip <filename>.zip <dir>
```
