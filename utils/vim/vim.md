[vim ](https://www.vim.org)

```sh
######## normal mode #########
[esc]     # 进入 normal mode
[u]       # undo，撤销
[ctrl][r] # 恢复上一步撤销的操作
[x]       # 删除光标后的一个字符
[X]       # 删除光标前的一个字符

[h]       # 向左移动
[j]       # 向下移动
[k]       # 向上移动
[l]       # 向右移动

[d][d]    # 删除当前行
[c]       # 修改
[c][c]    # 删除当前保留缩进（进入 insert）

# 跳转到
[$]             # 该行行尾
[0]             # 该行行首
[^]             # 该行行首第一个单词
[g][g]          # 文本开头
[G]             # 文件结尾
10[G]           # 第 10 行第一个单词

# 移动光标到
[f]a            # 右侧字符 a 上
[F]a            # 左侧字符 a 上
[t]a            # 右侧字符 a 之前
[T]a            # 左侧字符 a 之后

# 移动光标到
[w]             # 下一个单词开始
[e]             # 下一个单词字尾
[b]             # 上一个单词开始
[W]             # 下一个单词开始，不考虑符号
[E]             # 下一个单词结尾，不考虑符号
[B]             # 上一个单词开始，不考虑符号

# 复制
[v][y]          # 当前光标之后第一个字符
[y][y]          # 当前行
2yy             # 当前行向下 2 行（包含当前行）
[y][0]          # 当前光标之前到行首
[y][^]          # 当前光标之前到行首第一个字符
[y][$]          # 当前光标之后到行尾

# 切剪到
[D]             # 当前光标之后至行尾
[d][0]
[d][^]
[d][$]

[p]             # 向下粘贴
[P]             # 向下粘贴，光标不移动，光标之前

# 从光标（包含）删除到
[d][f]a         # 向右至第一个 a（包含）
[d][F]a         # 向左至第一个 a（包含）
[d][t]a         # 向右至第一个 a
[d][T]a         # 向左至第一个 a

[d][/]return    # 删除到 return
[C]             # 删除到当前行尾

# 从光标（包含）到（进入 insert）
[c][f]a         # 向右到 a 删除（包含），进行修改
[c][F]a         # 向左到 a 删除（包含），进行修改
[c][t]a         # 向右到 a 删除，进行修改
[c][T]a         # 向左到 a 删除，进行修改

[c][/]return    # 删除到 return

####### 选择模式 #########
[v] # 正常选择
[V] # 行选择


######## 输入模式 #########
i # 向光标位置输入
a # 向光标后一个位置输入
I # 行开头输入
A # 行末尾输入
o # 单独向下创建一行输入
O # 单独向上创建一行输入

######## 底行模式 #########
[shift][;] # 进入底行模式
q          # 退出
q!         # 强制退出（不保存当前修改）
w          # 保存
w!         # 强制保存
wq         # 保存并退出
wq!        # 强制保存并退出
set nu     # 显示行号
set nonu   # 不显示行号
: 20       # 跳转到第 20 行
set list   # 不可见字符
```

配置 vim

```sh
vim ~/.vimrc
```

```sh
set tabstop=4        # 设置制表符宽度为 4
set softtabstop=4    # 设置软制表符宽度为 4
set shiftwidth=4     # 设置缩进的空格数为 4
set autoindent       # 设置自动缩进
set nu               # 设置在左侧显示行号
set cindent          # 设置使用 c/c++ 语言的自动缩进方式

# 括号 引号补全
inoremap {  {}<esc>i
inoremap (  ()<esc>i
inoremap [  []<esc>i
inoremap '  ''<esc>i
inoremap "  ""<esc>i
```

`ctrl` `n` 或 `ctrl` `p` 来在当前文件中进行补全
