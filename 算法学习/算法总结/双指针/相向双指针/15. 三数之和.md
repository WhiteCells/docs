### [15. 三数之和](https://leetcode.cn/problems/3sum/)

给你一个整数数组 `nums` ，判断是否存在三元组 `[nums[i], nums[j], nums[k]]` 满足 `i != j`、`i != k` 且 `j != k` ，同时还满足 `nums[i] + nums[j] + nums[k] == 0` 。请

你返回所有和为 `0` 且不重复的三元组。

**注意：** 答案中不可以包含重复的三元组。

 

**示例 1：**

```
输入：nums = [-1,0,1,2,-1,-4]
输出：[[-1,-1,2],[-1,0,1]]
解释：
nums[0] + nums[1] + nums[2] = (-1) + 0 + 1 = 0 。
nums[1] + nums[2] + nums[4] = 0 + 1 + (-1) = 0 。
nums[0] + nums[3] + nums[4] = (-1) + 2 + (-1) = 0 。
不同的三元组是 [-1,0,1] 和 [-1,-1,2] 。
注意，输出的顺序和三元组的顺序并不重要。
```

**示例 2：**

```
输入：nums = [0,1,1]
输出：[]
解释：唯一可能的三元组和不为 0 。
```

**示例 3：**

```
输入：nums = [0,0,0]
输出：[[0,0,0]]
解释：唯一可能的三元组和为 0 。
```

 

**提示：**

- $3 <= nums.length <= 3000$
- $-10^5 <= nums[i] <= 10^5$

### 方法1

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> res;
        sort(nums.begin(), nums.end()); // O(nlogn)
        if (0 < nums[0]) { // 排序后第一个数大于 0, 则所有数之和不会等于 0
            return {};
        }
        for (int i = 0; i < nums.size() - 2; ++i) { // O(n)
            if (0 < i && nums[i] == nums[i - 1]) { // 当 nums[i] 重复出现时需要去重
                continue;
            }
            int j = i + 1, k = nums.size() - 1;
            while (j < k) { // O(n)
                if (i + 1 < j && nums[j] == nums[j - 1]) { // 当 nums[j] 重复出现时需要去重
                    ++j;
                    continue;
                }
                int sum = nums[i] + nums[j] + nums[k];
                if (0 < sum) {
                    --k;
                } else if (0 > sum) {
                    ++j;
                } else {
                    res.push_back({ nums[i], nums[j], nums[k] });
                    ++j;
                    --k;
                }
            }
        }
        return res;
    }
};
```

- 时间复杂度：$O(n^2)$
- 空间复杂度：$n(1)$

另一种写法

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int> &nums) {
        vector<vector<int>> res;
        sort(nums.begin(), nums.end());
        int n = nums.size();
        if (0 < nums[0]) { // nums 内全为正数, 则三数之和不会为 0
            return {};
        }
        for (int i = 0; i + 2 < n; ++i) { // nums.size() 最小为 3
            if (0 < i && nums[i] == nums[i - 1]) { // 当 nums[i] 重复出现时需要去重
                continue;
            }
            int j = i + 1, k = n - 1; // 左右指针
            while (j < k) {
                int sum = nums[i] + nums[j] + nums[k];
                if (0 == sum) {
                    res.push_back({ nums[i], nums[j], nums[k] });
                    ++j;
                    while (j > k && nums[j] == nums[j - 1]) {
                        ++j;
                    }
                    --k;
                    while (k > j && nums[k] == nums[k + 1]) {
                        --k;
                    }
                } else if (0 < sum) {
                    --k;
                } else {
                    ++j;
                }
            }
        }
        return res;
    }
};
```

再根据排序后的单调性可以得：

1. 如果 `nums[i] + nums[i+1] + nums[i+2] > 0` 则后面的任意三数之加都大于 0
2. 如果 `nums[i] + nums[nums.size()-2] + nums[nums.size()-1] < 0` 则 `nums[i]` 加后面的任意两个数小于 0，跳过 `i`

```cpp
class Solution {
public:
    vector<vector<int>> threeSum(vector<int>& nums) {
        vector<vector<int>> res;
        sort(nums.begin(), nums.end());
        if (0 < nums[0]) { // 排序后第一个数大于 0, 则所有数之和不会等于 0
            return {};
        }
        for (int i = 0; i < nums.size() - 2; ++i) {
            if (0 < i && nums[i] == nums[i - 1]) { // 当 nums[i] 重复出现时需要去重
                continue;
            }
            if (0 < nums[i] + nums[i + 1] + nums[i + 2]) { // 1
                break;
            }
            if (0 > nums[i] + nums[nums.size() - 2] + nums[nums.size() - 1]) { // 2
                continue;
            }
            int j = i + 1, k = nums.size() - 1;
            while (j < k) {
                if (i + 1 < j && nums[j] == nums[j - 1]) { // 当 nums[j] 重复出现时需要去重
                    ++j;
                    continue;
                }
                int sum = nums[i] + nums[j] + nums[k];
                if (0 < sum) {
                    --k;
                } else if (0 > sum) {
                    ++j;
                } else {
                    res.push_back({ nums[i], nums[j], nums[k] });
                    ++j;
                    --k;
                }
            }
        }
        return res; 
    }
};
```
