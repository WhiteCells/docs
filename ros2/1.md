拉取 `ros:humble` 镜像

```sh
docker pull ros:humble
```

使用 `ros:humble` 镜像构建容器

```sh
docker run -itd \
  --name ros2 \
  --hostname ros2 \
  --net=host \
  -v ~/ros2_ws:/root/ros2_ws \
  -v /tmp/.X11-unix:/tmp/.X11-unix \
  -e DISPLAY=$DISPLAY \
  ros:humble
```

进入容器

```sh
docker exec -it ros2 bash
```

激活 ros2 环境

```sh
apt update
apt install -y \
  ros-humble-turtlesim \
  ros-humble-rviz2 \
  x11-apps
source /opt/ros/humble/setup.sh
```



后续

```sh
docker pull ros:humble-desktop-full
docker run -itd \
  --name ros2 \
  --hostname ros2 \
  --net=host \
  --privileged \
  -e DISPLAY=$DISPLAY \
  -e QT_X11_NO_MITSHM=1 \
  -v /tmp/.X11-unix:/tmp/.X11-unix:rw \
  -v /dev:/dev \
  -v ~/ros2_ws:/root/ros2_ws \
  ros:humble-desktop
```

wayland

```sh
docker run -it \
  --name ros2 \
  --net=host \
  -e ROS_DOMAIN_ID=0 \
  -e WAYLAND_DISPLAY=$WAYLAND_DISPLAY \
  -e XDG_RUNTIME_DIR=$XDG_RUNTIME_DIR \
  -v $XDG_RUNTIME_DIR/$WAYLAND_DISPLAY:$XDG_RUNTIME_DIR/$WAYLAND_DISPLAY \
  ros:humble
```

# ArchLinux KDE Wayland Nvidia 安装 ros2

```sh
paru -S docker
paru -S nvidia-container-toolkit
sudo nvidia-ctk runtime configure --runtime=docker
sudo systemctl restart docker
paru -S xorg-xhost
xhost +local:
```

```sh
docker pull osrf/ros:lyrical-desktop
docker pull osrf/ros:lyrical-desktop-full
```

```sh
# 创建工作目录
mkdir -p ~/ros2_ws/src

docker run -dit \
    --name ros2 \
    --network host \
    --ipc host \
    --gpus all \
    -e DISPLAY=$DISPLAY \
    -e QT_QPA_PLATFORM=xcb \
    -v /tmp/.X11-unix:/tmp/.X11-unix \
    -v ~/ros2_ws:/root/ros2_ws \
    osrf/ros:lyrical-desktop

xhost +SI:localuser:root
```

进入容器后

```sh
export http_proxy=
export https_proxy=
source /opt/ros/lyrical/setup.bash
apt update
apt install -y mesa-utils x11-apps
rviz2
```
